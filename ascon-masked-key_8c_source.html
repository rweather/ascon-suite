28 void ascon_masked_key_128_init
29  (ascon_masked_key_128_t *masked, const unsigned char *key)
30 {
31  ascon_trng_state_t trng;
32  ascon_trng_init(&trng);
33 #if ASCON_MASKED_KEY_SHARES == 2
34  ascon_masked_word_x2_load(&(masked->k[0]), key, &trng);
35  ascon_masked_word_x2_load(&(masked->k[1]), key + 8, &trng);
36 #elif ASCON_MASKED_KEY_SHARES == 3
37  ascon_masked_word_x3_load(&(masked->k[0]), key, &trng);
38  ascon_masked_word_x3_load(&(masked->k[1]), key + 8, &trng);
39 #else
40  ascon_masked_word_x4_load(&(masked->k[0]), key, &trng);
41  ascon_masked_word_x4_load(&(masked->k[1]), key + 8, &trng);
42 #endif
43  ascon_trng_free(&trng);
44 }

90 void ascon_masked_key_160_init
91  (ascon_masked_key_160_t *masked, const unsigned char *key)
92 {
93  static unsigned char const zeroes[4] = {0, 0, 0, 0};
94  ascon_trng_state_t trng;
95  ascon_trng_init(&trng);
96 #if ASCON_MASKED_KEY_SHARES == 2
97  /* ASCON-80pq absorbs keys in two places so we need to mask it twice */
98  ascon_masked_word_x2_load(&(masked->k[0]), key, &trng);
99  ascon_masked_word_x2_load(&(masked->k[1]), key + 8, &trng);
100  ascon_masked_word_x2_load_32(&(masked->k[2]), key + 16, zeroes, &trng);
101  ascon_masked_word_x2_load_32(&(masked->k[3]), zeroes, key, &trng);
102  ascon_masked_word_x2_load(&(masked->k[4]), key + 4, &trng);
103  ascon_masked_word_x2_load(&(masked->k[5]), key + 12, &trng);
104 #elif ASCON_MASKED_KEY_SHARES == 3
105  ascon_masked_word_x3_load(&(masked->k[0]), key, &trng);
106  ascon_masked_word_x3_load(&(masked->k[1]), key + 8, &trng);
107  ascon_masked_word_x3_load_32(&(masked->k[2]), key + 16, zeroes, &trng);
108  ascon_masked_word_x3_load_32(&(masked->k[3]), zeroes, key, &trng);
109  ascon_masked_word_x3_load(&(masked->k[4]), key + 4, &trng);
110  ascon_masked_word_x3_load(&(masked->k[5]), key + 12, &trng);
111 #else
112  ascon_masked_word_x4_load(&(masked->k[0]), key, &trng);
113  ascon_masked_word_x4_load(&(masked->k[1]), key + 8, &trng);
114  ascon_masked_word_x4_load_32(&(masked->k[2]), key + 16, zeroes, &trng);
115  ascon_masked_word_x4_load_32(&(masked->k[3]), zeroes, key, &trng);
116  ascon_masked_word_x4_load(&(masked->k[4]), key + 4, &trng);
117  ascon_masked_word_x4_load(&(masked->k[5]), key + 12, &trng);
118 #endif
119  ascon_trng_free(&trng);
120 }

130 {

137 #elif ASCON_MASKED_KEY_SHARES == 3
138  for (index = 0; index < 6; ++index) {
139  ascon_masked_word_x2_randomize
140  (&(masked->k[index]), &(masked->k[index]), trng);
141  }
142 #else
143  for (index = 0; index < 6; ++index) {
144  ascon_masked_word_x2_randomize
145  (&(masked->k[index]), &(masked->k[index]), trng);
146  }
147 #endif
148 }
