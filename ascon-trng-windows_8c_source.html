34  * than CryptGenRandom() as it is more efficient than creating a
35  * cryptography service provider. But it is harder to access as
36  * there is no import library. Fix this later to dynamically load
37  * "Advapi32.dll" and resolve the entry point for RtlGenRandom(). */

39 int ascon_trng_generate(unsigned char *out, size_t outlen)
40 {
41  HCRYPTPROV provider = 0;
42  int ok = 0;
43  if (CryptAcquireContextW
44  (&provider, 0, 0, PROV_RSA_FULL,
45  CRYPT_VERIFYCONTEXT | CRYPT_SILENT)) {
46  if (CryptGenRandom(provider, outlen, out))
47  ok = 1;
48  CryptReleaseContext(provider, 0);
49  }
50  if (!ok) {
51  /* Could not open the provider or it didn't work; this is a problem! */
52  memset(out, 0, outlen);
53  }
54  return ok;
55 }

57 int ascon_trng_init(ascon_trng_state_t *state)
58 {
59  unsigned char seed[ASCON_SYSTEM_SEED_SIZE];
60  int ok = ascon_trng_generate(seed, sizeof(seed));
61  ascon_init(&(state->prng));
62  ascon_overwrite_bytes
63  (&(state->prng), seed, 40 - sizeof(seed), sizeof(seed));
64  ascon_permute12(&(state->prng));
65  ascon_release(&(state->prng));
66  ascon_clean(seed, sizeof(seed));
67  state->posn = 0;
68  return ok;
69 }

77 uint32_t ascon_trng_generate_32(ascon_trng_state_t *state)
78 {
79  uint32_t x;
80  ascon_acquire(&(state->prng));
81  if ((state->posn + sizeof(uint32_t)) > ASCON_TRNG_MIXER_RATE) {
82  ascon_permute6(&(state->prng));
83  state->posn = 0;
84  }
85 #if defined(ASCON_BACKEND_SLICED32) || defined(ASCON_BACKEND_SLICED64) || \
86  defined(ASCON_BACKEND_DIRECT_XOR)
87  /* Pull a word directly out of the state. It doesn't matter if the
88  * word is bit-sliced or not because any bit is as good as any other. */
89  x = state->prng.W[state->posn / sizeof(uint32_t)];
90 #else
91  ascon_extract_bytes
92  (&(state->prng), (unsigned char *)&x, state->posn, sizeof(x));
93 #endif
94  ascon_release(&(state->prng));
95  state->posn += sizeof(uint32_t);
96  return x;
97 }

99 uint64_t ascon_trng_generate_64(ascon_trng_state_t *state)
100 {
101  uint64_t x;
102  ascon_acquire(&(state->prng));
103  if ((state->posn + sizeof(uint64_t)) > ASCON_TRNG_MIXER_RATE ||
104  (state->posn % 8U) != 0) {
105  ascon_permute6(&(state->prng));
106  state->posn = 0;
107  }
108 #if defined(ASCON_BACKEND_SLICED32) || defined(ASCON_BACKEND_SLICED64) || \
109  defined(ASCON_BACKEND_DIRECT_XOR)
110  /* Pull a word directly out of the state. It doesn't matter if the
111  * word is bit-sliced or not because any bit is as good as any other. */
112  x = state->prng.S[state->posn / sizeof(uint64_t)];
113 #else
114  ascon_extract_bytes
115  (&(state->prng), (unsigned char *)&x, state->posn, sizeof(x));
116 #endif
117  ascon_release(&(state->prng));
118  state->posn += sizeof(uint64_t);
119  return x;
120 }

122 int ascon_trng_reseed(ascon_trng_state_t *state)
123 {
124  unsigned char seed[ASCON_SYSTEM_SEED_SIZE];
125  int ok = ascon_trng_generate(seed, sizeof(seed));
126  ascon_acquire(&(state->prng));
127  ascon_add_bytes(&(state->prng), seed, 40 - sizeof(seed), sizeof(seed));
128  ascon_overwrite_with_zeroes(&(state->prng), 0, 8); /* Forward security */
129  ascon_permute12(&(state->prng));
130  ascon_release(&(state->prng));
131  ascon_clean(seed, sizeof(seed));
132  state->posn = 0;
133  return ok;
134 }
