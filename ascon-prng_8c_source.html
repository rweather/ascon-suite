51 static void ascon_random_rekey(ascon_random_state_t *state)
52 {
53  int temp;
54 
55  /* Zero out part of the state and run the permutation several times.
56  * This enforces forward security on the SpongePRNG state. */
57  for (temp = 0; temp < (40 - ASCON_XOF_RATE); temp += ASCON_XOF_RATE)
58  ascon_xof_clear_rate(&(state->xof));

85 void ascon_random_generate
86  (ascon_random_state_t *state, unsigned char *out, size_t outlen)
87 {
88  /* If there is no state, use the global ascon_random() function
89  * so that we return something. Safer than returning nothing
90  * to the caller by accident and having them use that nothing. */
91  if (!state) {
92  ascon_random(out, outlen);
93  return;
94  }
95 
96  /* Force a re-seed if we have generated too many bytes so far */
97  if (state->counter >= ASCON_RANDOM_RESEED_LIMIT)
98  ascon_random_reseed(state);
99 
100  /* Squeeze data out of the PRNG state */
101  ascon_xof_squeeze(&(state->xof), out, outlen);
102  if (outlen < ASCON_RANDOM_RESEED_LIMIT)
103  state->counter += outlen;
104  else
105  state->counter = ASCON_RANDOM_RESEED_LIMIT;
106 
107  /* Re-key the PRNG to enforce forward security */
108  ascon_random_rekey(state);
109 }

111 int ascon_random_reseed(ascon_random_state_t *state)
112 {
113  if (state) {
114  /* Generate a new system seed and absorb it into the state */
115  unsigned char seed[ASCON_SYSTEM_SEED_SIZE];
116  int ok = ascon_trng_generate(seed, sizeof(seed));
117  ascon_xof_absorb(&(state->xof), seed, sizeof(seed));
118  ascon_clean(seed, sizeof(seed));
119 
120  /* Reset the re-seed counter to 0 */
121  state->counter = 0;
122 
123  /* Re-key the PRNG to enforce forward security */
124  ascon_random_rekey(state);
125  return ok;
126  }
127  return 0;
128 }

139 void ascon_random_add_entropy_quick
140  (ascon_random_state_t *state, uint64_t entropy)
141 {
142  if (state) {
143  /* Force the XOF object back into absorbing mode if necessary */
144  ascon_xof_pad(&(state->xof));
145 
146  /* Incorporate the bits into the state as quickly as we can.
147  * The bit order isn't important. One random bit is just as
148  * good as another so we directly XOR the entropy in. */
149  ascon_acquire(&(state->xof.state));
150 #if defined(ASCON_BACKEND_SLICED64) || defined(ASCON_BACKEND_SLICED32) || \
151  defined(ASCON_BACKEND_DIRECT_XOR)
152  state->xof.state.S[0] ^= entropy;
153 #else
154  ascon_add_bytes
155  (&(state->xof.state), (const unsigned char *)&entropy, 0,
156  sizeof(entropy));
157 #endif
158 
159  /* Perform two rounds of ASCON to mix in the entropy */
160  ascon_permute(&(state->xof.state), 10); /* Start at round 10 of 12 */
161  ascon_release(&(state->xof.state));
162  }
163 }
