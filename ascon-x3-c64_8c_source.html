51 /* Generate a pre-inverted round constant so that we can
52  * avoid NOT'ing x2 in the S-box during the rounds */
53 #define ROUND_CONSTANT(round) \
54  (~(uint64_t)(((0x0F - (round)) << 4) | (round)))

56 void ascon_x3_permute
57  (ascon_masked_state_t *state, uint8_t first_round, uint64_t preserve[2])
58 {
59  static const uint64_t RC[12] = {
60  ROUND_CONSTANT(0),
61  ROUND_CONSTANT(1),
62  ROUND_CONSTANT(2),
63  ROUND_CONSTANT(3),
64  ROUND_CONSTANT(4),
65  ROUND_CONSTANT(5),
66  ROUND_CONSTANT(6),
67  ROUND_CONSTANT(7),
68  ROUND_CONSTANT(8),
69  ROUND_CONSTANT(9),
70  ROUND_CONSTANT(10),
71  ROUND_CONSTANT(11)
72  };
73  uint64_t x0_a, x1_a, x2_a, x3_a, x4_a;
74  uint64_t x0_b, x1_b, x2_b, x3_b, x4_b;
75  uint64_t x0_c, x1_c, x2_c, x3_c, x4_c;
76  uint64_t t0_a, t0_b, t0_c, t1_a, t1_b, t1_c;
77 
78  /* Start with the randomness that the caller provided */
79  t0_a = preserve[0];
80  t0_b = preserve[1];
81 
82  /* Load the state into local variables */
83  x0_a = state->M[0].S[0];
84  x0_b = state->M[0].S[1];
85  x0_c = state->M[0].S[2];
86  x1_a = state->M[1].S[0];
87  x1_b = state->M[1].S[1];
88  x1_c = state->M[1].S[2];
89  x2_a = state->M[2].S[0];
90  x2_b = state->M[2].S[1];
91  x2_c = state->M[2].S[2];
92  x3_a = state->M[3].S[0];
93  x3_b = state->M[3].S[1];
94  x3_c = state->M[3].S[2];
95  x4_a = state->M[4].S[0];
96  x4_b = state->M[4].S[1];
97  x4_c = state->M[4].S[2];
98 
99  /* The round constants invert x2 as part of the rounds so that we
100  * don't need an explicit "x2 = ~x2" step in the S-box. Pre-invert
101  * x2 before the first round to compensate. */
102  x2_a = ~x2_a;
103 
104  /* Perform all encryption rounds */
105  while (first_round < 12) {
106  /* Add the inverted round constant to x2 */
107  x2_a ^= RC[first_round++];
108 
109  /* Start of the substitution layer, first share */
110  x0_a ^= x4_a;
111  x4_a ^= x3_a;
112  x2_a ^= x1_a;
113  t1_a = x0_a;
114 
115  /* Start of the substitution layer, second share */
116  x0_b ^= x4_b;
117  x4_b ^= x3_b;
118  x2_b ^= x1_b;
119  t1_b = x0_b;
120 
121  /* Start of the substitution layer, third share */
122  x0_c ^= x4_c;
123  x4_c ^= x3_c;
124  x2_c ^= x1_c;
125  t1_c = x0_c;
126 
127  /* Middle part of the substitution layer, Chi5 */
128  t0_c = ascon_mask64_rotate_share2_0(t0_a) ^ /* t0 = random shares */
129  ascon_mask64_rotate_share2_1(t0_b);
130  and_not_xor(t0, x0, x1); /* t0 ^= (~x0) & x1; */
131  and_not_xor(x0, x1, x2); /* x0 ^= (~x1) & x2; */
132  and_not_xor(x1, x2, x3); /* x1 ^= (~x2) & x3; */
133  and_not_xor(x2, x3, x4); /* x2 ^= (~x3) & x4; */
134  and_not_xor(x3, x4, t1); /* x3 ^= (~x4) & t1; */
135  x4_a ^= t0_a; /* x4 ^= t0; */
136  x4_b ^= t0_b;
137  x4_c ^= t0_c;
138 
139  /* End of the substitution layer */
140  x1_a ^= x0_a;
141  x0_a ^= x4_a;
142  x3_a ^= x2_a;
143  x1_b ^= x0_b;
144  x0_b ^= x4_b;
145  x3_b ^= x2_b;
146  x1_c ^= x0_c;
147  x0_c ^= x4_c;
148  x3_c ^= x2_c;
149 
150  /* NOT'ing x2 is done as part of the next round constant */
151  /* x2_a = ~x2_a; */
152 
153  /* Linear diffusion layer, third share */
154  x0_c ^= rightRotate19_64(x0_c) ^ rightRotate28_64(x0_c);
155  x1_c ^= rightRotate61_64(x1_c) ^ rightRotate39_64(x1_c);
156  x2_c ^= rightRotate1_64(x2_c) ^ rightRotate6_64(x2_c);
157  x3_c ^= rightRotate10_64(x3_c) ^ rightRotate17_64(x3_c);
158  x4_c ^= rightRotate7_64(x4_c) ^ rightRotate41_64(x4_c);
159 
160  /* Linear diffusion layer, second share */
161  x0_b ^= rightRotate19_64(x0_b) ^ rightRotate28_64(x0_b);
162  x1_b ^= rightRotate61_64(x1_b) ^ rightRotate39_64(x1_b);
163  x2_b ^= rightRotate1_64(x2_b) ^ rightRotate6_64(x2_b);
164  x3_b ^= rightRotate10_64(x3_b) ^ rightRotate17_64(x3_b);
165  x4_b ^= rightRotate7_64(x4_b) ^ rightRotate41_64(x4_b);
166 
167  /* Linear diffusion layer, first share */
168  x0_a ^= rightRotate19_64(x0_a) ^ rightRotate28_64(x0_a);
169  x1_a ^= rightRotate61_64(x1_a) ^ rightRotate39_64(x1_a);
170  x2_a ^= rightRotate1_64(x2_a) ^ rightRotate6_64(x2_a);
171  x3_a ^= rightRotate10_64(x3_a) ^ rightRotate17_64(x3_a);
172  x4_a ^= rightRotate7_64(x4_a) ^ rightRotate41_64(x4_a);
173 
174  /* Rotate the randomness in t0 before the next round */
175  t0_a = rightRotate13_64(t0_a);
176  t0_b = rightRotate29_64(t0_b);
177  }
178 
179  /* Return the final randomness to the caller to preserve it */
180  preserve[0] = t0_a;
181  preserve[1] = t0_b;
182 
183  /* Store the local variables back to the state with a final invert of x2 */
184  state->M[0].S[0] = x0_a;
185  state->M[0].S[1] = x0_b;
186  state->M[0].S[2] = x0_c;
187  state->M[1].S[0] = x1_a;
188  state->M[1].S[1] = x1_b;
189  state->M[1].S[2] = x1_c;
190  state->M[2].S[0] = ~x2_a;
191  state->M[2].S[1] = x2_b;
192  state->M[2].S[2] = x2_c;
193  state->M[3].S[0] = x3_a;
194  state->M[3].S[1] = x3_b;
195  state->M[3].S[2] = x3_c;
196  state->M[4].S[0] = x4_a;
197  state->M[4].S[1] = x4_b;
198  state->M[4].S[2] = x4_c;
199 }
