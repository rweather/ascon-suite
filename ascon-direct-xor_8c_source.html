41 void ascon_init(ascon_state_t *state)
42 {
43 #if defined(ASCON_CHECK_ACQUIRE_RELEASE)
44  if (acquired) {
45  fprintf(stderr, "acquire and release operations are not balanced\n");
46  abort();
47  }
48  acquired = 1;
49 #endif
