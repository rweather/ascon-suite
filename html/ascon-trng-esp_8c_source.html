37 int ascon_trng_generate(unsigned char *out, size_t outlen)
38 {
39  uint32_t x;
40  while (outlen >= sizeof(x)) {
41  x = esp_random();
42  memcpy(out, &x, sizeof(x));
43  out += sizeof(x);
44  outlen -= sizeof(x);
45  }
46  if (outlen > 0) {
47  x = esp_random();
48  memcpy(out, &x, outlen);
49  }
50  return 1; /* Assume that it works */
51 }
