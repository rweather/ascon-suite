26 void ascon_xofa(unsigned char *out, const unsigned char *in, size_t inlen)
27 {
28  ascon_xofa_state_t state;
29  ascon_xofa_init(&state);
30  ascon_xofa_absorb(&state, in, inlen);
31  ascon_xofa_squeeze(&state, out, ASCON_HASH_SIZE);
32  ascon_xofa_free(&state);
33 }

73 {
74 #if !defined(__SIZEOF_SIZE_T__) || __SIZEOF_SIZE_T__ >= 4
75  if (outlen >= (((size_t)1) << 29))
76  outlen = 0; /* Too large, so switch to arbitrary-length output */

161 void ascon_xofa_absorb
162  (ascon_xofa_state_t *state, const unsigned char *in, size_t inlen)
163 {
164  unsigned temp;
165 
166  /* Acquire access to shared hardware if necessary */
167  ascon_acquire(&(state->state));
168 
169  /* If we were squeezing output, then go back to the absorb phase */
170  if (state->mode) {
171  state->mode = 0;
172  state->count = 0;
173  ascon_permute(&(state->state), 0);
174  }
175 
176  /* Handle the partial left-over block from last time */
177  if (state->count) {
178  temp = ASCON_XOF_RATE - state->count;
179  if (temp > inlen) {
180  temp = (unsigned)inlen;
181  ascon_absorb_partial(&(state->state), in, state->count, temp);
182  state->count += temp;
183  ascon_release(&(state->state));
184  return;
185  }
186  ascon_absorb_partial(&(state->state), in, state->count, temp);
187  state->count = 0;
188  in += temp;
189  inlen -= temp;
190  ascon_permute(&(state->state), 4);
191  }
192 
193  /* Process full blocks that are aligned at state->s.count == 0 */
194  while (inlen >= ASCON_XOF_RATE) {
195  ascon_absorb_8(&(state->state), in, 0);
196  in += ASCON_XOF_RATE;
197  inlen -= ASCON_XOF_RATE;
198  ascon_permute(&(state->state), 4);
199  }
200 
201  /* Process the left-over block at the end of the input */
202  temp = (unsigned)inlen;
203  if (temp > 0)
204  ascon_absorb_partial(&(state->state), in, 0, temp);
205  state->count = temp;
206 
207  /* Release access to the shared hardware */
208  ascon_release(&(state->state));
209 }

211 void ascon_xofa_squeeze
212  (ascon_xofa_state_t *state, unsigned char *out, size_t outlen)
213 {
214  unsigned temp;
215 
216  /* Acquire access to shared hardware if necessary */
217  ascon_acquire(&(state->state));
218 
219  /* Pad the final input block if we were still in the absorb phase */
220  if (!state->mode) {
221  ascon_pad(&(state->state), state->count);
222  state->count = 0;
223  state->mode = 1;
224  ascon_permute(&(state->state), 0);
225  }
226 
227  /* Handle left-over partial blocks from last time */
228  if (state->count) {
229  temp = ASCON_XOF_RATE - state->count;
230  if (temp > outlen) {
231  temp = (unsigned)outlen;
232  ascon_squeeze_partial(&(state->state), out, state->count, temp);
233  state->count += temp;
234  ascon_release(&(state->state));
235  return;
236  }
237  ascon_squeeze_partial(&(state->state), out, state->count, temp);
238  out += temp;
239  outlen -= temp;
240  state->count = 0;
241  ascon_permute(&(state->state), 4);
242  }
243 
244  /* Handle full blocks */
245  while (outlen >= ASCON_XOF_RATE) {
246  ascon_squeeze_8(&(state->state), out, 0);
247  ascon_permute(&(state->state), 4);
248  out += ASCON_XOF_RATE;
249  outlen -= ASCON_XOF_RATE;
250  }
251 
252  /* Handle the left-over block */
253  if (outlen > 0) {
254  temp = (unsigned)outlen;
255  ascon_squeeze_partial(&(state->state), out, 0, temp);
256  state->count = temp;
257  }
258 
259  /* Release access to the shared hardware */
260  ascon_release(&(state->state));
261 }

263 void ascon_xofa_pad(ascon_xofa_state_t *state)
264 {
265  if (state->mode) {
266  /* We were squeezing output, so re-enter the absorb phase
267  * which will implicitly align on a rate block boundary */
268  ascon_xofa_absorb(state, 0, 0);
269  } else if (state->count != 0) {
270  /* Not currently aligned, so invoke the permutation */
271  ascon_acquire(&(state->state));
272  ascon_permute(&(state->state), 4);
273  ascon_release(&(state->state));
274  state->count = 0;
275  }
276 }
