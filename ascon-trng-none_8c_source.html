107 /* Add the current time to the PRNG state and then re-key the state */
108 static void ascon_trng_add_time(ascon_state_t *state)
109 {
110 #if defined(ARDUINO)
111  /* Add the current Arduino time as a seed to provide some extra jitter */
112  {
113  uint32_t x[2];
114  x[0] = (uint32_t)millis();
115  x[1] = (uint32_t)micros();
116  ascon_add_bytes(state, (const unsigned char *)x, 0, sizeof(x));
117  }
118 #elif defined(USE_HAL_DRIVER)
119  /* Mix in the STM32 millisecond tick counter for some extra jitter */
120  {
121  uint32_t x = HAL_GetTick();
122  ascon_add_bytes(state, (const unsigned char *)&x, 0, sizeof(x));
123  }
124 #elif defined(HAVE_CLOCK_GETTIME)
125  /* Mix in the monotonic and real times in nanoseconds */
126  {
127  struct timespec ts;
128 #if defined(CLOCK_MONOTONIC)
129  clock_gettime(CLOCK_MONOTONIC, &ts);
130  ascon_trng_add_timespec
131  (state, 0, (uint32_t)(ts.tv_sec), (uint32_t)(ts.tv_nsec));
132 #endif
133  clock_gettime(CLOCK_REALTIME, &ts);
134  ascon_trng_add_timespec
135  (state, 8, (uint32_t)(ts.tv_sec), (uint32_t)(ts.tv_nsec));
136  ascon_clean(&ts, sizeof(ts));
137  }
138 #elif defined(HAVE_GETTIMEOFDAY)
139  /* Mix in the current time of day in microseconds */
140  {
141  struct timeval tv;
142  gettimeofday(&tv, 0);
143  ascon_trng_add_timespec
144  (state, 0, (uint32_t)(tv.tv_sec), (uint32_t)(tv.tv_usec));
145  ascon_clean(&tv, sizeof(tv));
146  }
147 #elif defined(HAVE_TIME)
148  /* Mix in the current time of day in seconds (very little jitter) */
149  ascon_trng_add_timespec(state, 0, (uint32_t)time(0), 0);
150 #endif
151 
152  /* Permute the state for 6 rounds */
153  ascon_permute6(state);
154 
155  /* Zero out part of the state to provide forward security */
156  ascon_overwrite_with_zeroes(state, 0, 8);
157 
158  /* Permute the state for another 6 rounds */
159  ascon_permute6(state);
160 }

162 /* Squeeze data out of a PRNG */
163 static void ascon_trng_squeeze
164  (ascon_state_t *state, unsigned char *out, size_t outlen)
165 {
166  while (outlen >= 8U) {
167  ascon_extract_bytes(state, out, 0, 8);
168  ascon_permute6(state);
169  out += 8;
170  outlen -= 8;
171  }
172  if (outlen > 0U) {
173  ascon_extract_bytes(state, out, 0, outlen);
174  ascon_permute6(state);
175  }
176 }

178 /* Make sure that the global PRNG is initialized and seeded */
179 static int ascon_trng_global_init(unsigned char seed[ASCON_SYSTEM_SEED_SIZE])
180 {
181  int ok = 0;
182 
183  /* Acquire access to the global PRNG object */
184  if (!global_prng_initialized) {
185  global_prng_initialized = 1;
186  ascon_init(&global_prng);
187  } else {
188  ascon_acquire(&global_prng);
189  }
190 
191  /* See if the application is willing to give us TRNG data */
192  if (ascon_trng_get_bytes(seed, ASCON_SYSTEM_SEED_SIZE)) {
193  ascon_add_bytes(&global_prng, seed, 8, ASCON_SYSTEM_SEED_SIZE);
194  ok = 1;
195  }
196 
197  /* Add the current time to the global PRNG and re-key */
198  ascon_trng_add_time(&global_prng);
199  return ok;
200 }

202 int ascon_trng_generate(unsigned char *out, size_t outlen)
203 {
204  unsigned char seed[ASCON_SYSTEM_SEED_SIZE];
205  int ok;
206 
207  /* If the application has declared ascon_trng_get_bytes() to be good,
208  * then use it directly rather than run a global PRNG. We fall through
209  * if ascon_trng_get_bytes() subsequently fails anyway. */
210  if (ascon_trng_get_bytes_is_good()) {
211  ok = ascon_trng_get_bytes(out, outlen);
212  if (ok)
213  return 1;
214  }
215 
216  /* Re-seed and squeeze some data out of the global PRNG */
217  ok = ascon_trng_global_init(seed);
218  ascon_trng_squeeze(&global_prng, out, outlen);
219 
220  /* Re-key the global PRNG after generating the bytes */
221  ascon_overwrite_with_zeroes(&global_prng, 0, 8);
222  ascon_permute6(&global_prng);
223  ascon_release(&global_prng);
224  ascon_clean(seed, sizeof(seed));
225  return ok;
226 }

228 int ascon_trng_init(ascon_trng_state_t *state)
229 {
230  unsigned char seed[ASCON_SYSTEM_SEED_SIZE];
231  int ok = 0;
232 
233  /* Try getting the seed from the application first if its data is good */
234  if (ascon_trng_get_bytes_is_good()) {
235  ok = ascon_trng_get_bytes(seed, sizeof(seed));
236  }
237  if (!ok) {
238  /* Intialize and re-seed the global PRNG */
239  ok = ascon_trng_global_init(seed);
240 
241  /* Squeeze out a seed value to use to initialize "state" */
242  ascon_trng_squeeze(&global_prng, seed, sizeof(seed));
243 
244  /* Re-key the global PRNG to enforce forward security */
245  ascon_overwrite_with_zeroes(&global_prng, 0, 8);
246  ascon_permute6(&global_prng);
247  ascon_release(&global_prng);
248  }
249 
250  /* Set up the new TRNG state */
251  ascon_init(&(state->prng));
252  ascon_add_bytes(&(state->prng), seed, 40 - sizeof(seed), sizeof(seed));
253  ascon_permute12(&(state->prng));
254  ascon_release(&(state->prng));
255  ascon_clean(seed, sizeof(seed));
256  state->posn = 0;
257  return ok;
258 }

266 uint32_t ascon_trng_generate_32(ascon_trng_state_t *state)
267 {
268  uint32_t x;
269  ascon_acquire(&(state->prng));
270  if ((state->posn + sizeof(uint32_t)) > ASCON_TRNG_MIXER_RATE) {
271  ascon_permute6(&(state->prng));
272  state->posn = 0;
273  }
274 #if defined(ASCON_BACKEND_SLICED32) || defined(ASCON_BACKEND_SLICED64) || \
275  defined(ASCON_BACKEND_DIRECT_XOR)
276  /* Pull a word directly out of the state. It doesn't matter if the
277  * word is bit-sliced or not because any bit is as good as any other. */
278  x = state->prng.W[state->posn / sizeof(uint32_t)];
279 #else
280  ascon_extract_bytes
281  (&(state->prng), (unsigned char *)&x, state->posn, sizeof(x));
282 #endif
283  ascon_release(&(state->prng));
284  state->posn += sizeof(uint32_t);
285  return x;
286 }

288 uint64_t ascon_trng_generate_64(ascon_trng_state_t *state)
289 {
290  uint64_t x;
291  ascon_acquire(&(state->prng));
292  if ((state->posn + sizeof(uint64_t)) > ASCON_TRNG_MIXER_RATE ||
293  (state->posn % 8U) != 0) {
294  ascon_permute6(&(state->prng));
295  state->posn = 0;
296  }
297 #if defined(ASCON_BACKEND_SLICED32) || defined(ASCON_BACKEND_SLICED64) || \
298  defined(ASCON_BACKEND_DIRECT_XOR)
299  /* Pull a word directly out of the state. It doesn't matter if the
300  * word is bit-sliced or not because any bit is as good as any other. */
301  x = state->prng.S[state->posn / sizeof(uint64_t)];

305 #endif
306  ascon_release(&(state->prng));
307  state->posn += sizeof(uint64_t);

311 int ascon_trng_reseed(ascon_trng_state_t *state)
312 {
313  unsigned char seed[ASCON_SYSTEM_SEED_SIZE];
314  int ok = 0;
315 
316  /* If the application has declared ascon_trng_get_bytes() to be good,
317  * then use it directly rather than run a global PRNG. We fall through
318  * if ascon_trng_get_bytes() subsequently fails anyway. */
319  if (ascon_trng_get_bytes_is_good()) {
320  ok = ascon_trng_get_bytes(seed, sizeof(seed));
321  }
322  if (!ok) {
323  /* Get a new key for the local PRNG from the global PRNG */
324  ok = ascon_trng_global_init(seed);
325  ascon_trng_squeeze(&global_prng, seed, ASCON_SYSTEM_SEED_SIZE);
326 
327  /* Re-key the global PRNG */
328  ascon_overwrite_with_zeroes(&global_prng, 0, 8);
329  ascon_permute6(&global_prng);
330  ascon_release(&global_prng);
331  }
332 
333  /* Re-key the local PRNG with the seed data from the global PRNG */
334  ascon_acquire(&(state->prng));
335  ascon_add_bytes(&(state->prng), seed, 8, ASCON_SYSTEM_SEED_SIZE);
336  ascon_permute6(&(state->prng));
337  ascon_overwrite_with_zeroes(&(state->prng), 0, 8);
338  ascon_permute6(&(state->prng));
339  ascon_clean(seed, sizeof(seed));
340  state->posn = 0;
341  return ok;
342 }
